// --- API Helpers ---

const callGemini = async (prompt, systemInstruction = "") => {
  try {
    const payload = {
      contents: [{ parts: [{ text: prompt }] }],
    };

    if (systemInstruction) {
      payload.systemInstruction = { parts: [{ text: systemInstruction }] };
    }

    const response = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`,
      {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload),
      }
    );

    if (!response.ok) throw new Error("Gemini API Error");
    const data = await response.json();
    return data.candidates?.[0]?.content?.parts?.[0]?.text || "No response";
  } catch (error) {
    console.error("Gemini Error:", error);
    return null;
  }
};

// --- Custom SVG Price Chart Component ---
const PriceChart = ({ history }) => {
  const maxPrice = Math.max(...history.map(d => d.price)) * 1.05;
  const minPrice = Math.min(...history.map(d => d.price)) * 0.95;
  const range = maxPrice - minPrice;
  const width = 100;
  const height = 50;

  const points = history.map((point, index) => {
    const x = (index / (history.length - 1)) * width;
    const y = height - ((point.price - minPrice) / range) * height;
    return `${x},${y}`;
  }).join(' ');

  const areaPath = `${points} ${width},${height} 0,${height}`;

  return (
    <div className="w-full">
      <div className="relative aspect-[2/1] w-full">
        <svg viewBox={`0 0 ${width} ${height}`} className="w-full h-full overflow-visible">
          <defs>
            <linearGradient id="chartGradient" x1="0" x2="0" y1="0" y2="1">
              <stop offset="0%" stopColor="#818cf8" stopOpacity="0.5" />
              <stop offset="100%" stopColor="#818cf8" stopOpacity="0" />
            </linearGradient>
          </defs>
          <line x1="0" y1="0" x2="100" y2="0" stroke="#334155" strokeWidth="0.5" strokeDasharray="2" />
          <line x1="0" y1="25" x2="100" y2="25" stroke="#334155" strokeWidth="0.5" strokeDasharray="2" />
          <line x1="0" y1="50" x2="100" y2="50" stroke="#334155" strokeWidth="0.5" strokeDasharray="2" />
          <polygon points={areaPath} fill="url(#chartGradient)" />
          <polyline points={points} fill="none" stroke="#818cf8" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" />
          {history.map((point, index) => {
             const x = (index / (history.length - 1)) * width;
             const y = height - ((point.price - minPrice) / range) * height;
             return (
               <g key={index} className="group">
                 <circle cx={x} cy={y} r="3" className="fill-indigo-500 stroke-slate-900 stroke-2" />
                 <foreignObject x={x - 15} y={y - 20} width="30" height="20" className="opacity-0 group-hover:opacity-100 transition-opacity overflow-visible">
                    <div className="bg-slate-800 text-[8px] text-white px-1 py-0.5 rounded shadow border border-slate-600 whitespace-nowrap -translate-y-full -translate-x-1/4">
                       â‚¹{(point.price/1000).toFixed(1)}k
                    </div>
                 </foreignObject>
               </g>
             );
          })}
        </svg>
      </div>
      <div className="flex justify-between mt-2 px-1">
        {history.map((h, i) => (
          <span key={i} className="text-xs text-slate-500">{h.date}</span>
        ))}
      </div>
    </div>
  );
};

  const [chatStore, setChatStore] = useState(null);
  const [chatMessage, setChatMessage] = useState("");
  
  // AI State
  const [aiRecommendations, setAiRecommendations] = useState([]);
  const [qaInput, setQaInput] = useState("");
  const [qaResponse, setQaResponse] = useState("");
  const [qaLoading, setQaLoading] = useState(false);

  // Review Summary State
  const [reviewSummary, setReviewSummary] = useState("");
  const [isGeneratingSummary, setIsGeneratingSummary] = useState(false);

  // Helper to simulate network delay for non-AI parts
  const simulateLoading = (nextView, duration = 1000, msg = "Loading...") => {
    setLoading(true);
    setLoadingMsg(msg);
    setTimeout(() => {
      setLoading(false);
      setView(nextView);
    }, duration);
  };

  const showNotification = (msg) => {
    setNotification(msg);
    setTimeout(() => setNotification(null), 3000);
  };

  const toggleFavorite = (id) => {
    if (favorites.includes(id)) {
      setFavorites(favorites.filter(fid => fid !== id));
      showNotification("Removed from favorites");
    } else {
      setFavorites([...favorites, id]);
      showNotification("Added to favorites");
    }
  };

  // --- Gemini Functions ---

  const handleAIRecommendation = async () => {
    setLoading(true);
    setLoadingMsg("Asking Gemini to find your best matches...");

    const inventoryString = JSON.stringify(MOCK_MOBILES.map(m => ({ id: m.id, name: m.name, features: m.features, specs: m.specs, price: m.price })));
    const userNeeds = `Selected Tags: ${selectedFeatures.join(', ')}. User Notes: "${customFeatureQuery}"`;

    const prompt = `
      Act as a mobile phone expert. 
      I have this inventory: ${inventoryString}.
      
      The user wants: ${userNeeds}.
      
      Task:
      1. Analyze the inventory against the user needs.
      2. Select the top 1-3 best matching phones.
      3. Return a JSON ARRAY of objects. Each object must have:
         - "id": (number) matching the inventory ID.
         - "matchReason": (string) A short, punchy sentence explaining why this specific phone fits their specific request.
      
      Return ONLY valid JSON. Do not use markdown blocks.
    `;

    try {
      const result = await callGemini(prompt);
      let recommendations = [];
      
      if (result) {
        const cleanJson = result.replace(/```json|```/g, '').trim();
        const parsedIds = JSON.parse(cleanJson);
        recommendations = parsedIds.map(rec => {
          const product = MOCK_MOBILES.find(m => m.id === rec.id);
          return product ? { ...product, matchReason: rec.matchReason } : null;
        }).filter(Boolean);
      }

      if (recommendations.length === 0) {
        recommendations = MOCK_MOBILES.slice(0, 3).map(m => ({...m, matchReason: "A popular choice based on your tags."}));
      }

      setAiRecommendations(recommendations);
      setLoading(false);
      setView('ai-suggestions');

    } catch (e) {
      console.error("AI matching failed", e);
      setAiRecommendations(MOCK_MOBILES.slice(0, 3).map(m => ({...m, matchReason: "Recommended based on general popularity."})));
      setLoading(false);
      setView('ai-suggestions');
    }
  };

  const handleProductQA = async () => {
    if (!qaInput.trim() || !selectedProduct) return;
    setQaLoading(true);
    const prompt = `
      Context: User is looking at a phone named "${selectedProduct.name}".
      Specs: ${JSON.stringify(selectedProduct)}.
      User Question: "${qaInput}"
      Answer the user's question accurately based on the specs provided. 
      Keep it brief (under 2 sentences). Be helpful and enthusiastic.
    `;
    const answer = await callGemini(prompt);
    setQaResponse(answer || "Sorry, I couldn't connect to the AI at the moment.");
    setQaLoading(false);
  };

  const handleReviewSummary = async () => {
    if (!selectedProduct) return;
    setIsGeneratingSummary(true);
    const prompt = `
      Write a short, balanced summary of user reviews for a smartphone called "${selectedProduct.name}".
      Assume it has a rating of ${selectedProduct.rating} stars.
      Highlight 3 specific pros (e.g. Camera, Battery) and 1 con.
      Use bullet points. Keep it under 60 words.
    `;
    const summary = await callGemini(prompt);
    setReviewSummary(summary || "Unable to generate summary at this time.");
    setIsGeneratingSummary(false);
  };

  const startWhatsAppChat = (store, message) => {
    setChatStore(store);
    setChatMessage(message);
    simulateLoading('whatsapp-chat', 600, `Connecting to ${store.name} on WhatsApp...`);
  };

  // --- Layout Components ---

  const BottomNav = () => {
    if (view === 'login' || view === 'checkout' || view === 'whatsapp-chat') return null;

    const navItems = [
      { id: 'home', icon: Home, label: 'Home' },
      { id: 'results', icon: Search, label: 'Search' }, 
      { id: 'compare', icon: ArrowRightLeft, label: 'Compare' },
      { id: 'profile', icon: User, label: 'Profile' }
    ];

    const handleNavClick = (id) => {
        if (id === 'home') setView('home');
        if (id === 'results') {
             if (searchQuery) setView('results');
             else setView('home');
        }
        if (id === 'compare') setView('ai-suggestions');
        if (id === 'profile') showNotification("Profile feature coming soon!");
    };

    return (
      <div className="absolute bottom-0 left-0 right-0 bg-slate-900 border-t border-slate-800 px-6 py-3 flex justify-between items-center z-50 shadow-[0_-5px_10px_rgba(0,0,0,0.3)]">
        {navItems.map((item) => (
          <button 
            key={item.id}
            onClick={() => handleNavClick(item.id)}
            className={`flex flex-col items-center gap-1 ${view === item.id || (item.id === 'results' && view === 'results') ? 'text-indigo-400' : 'text-slate-500 hover:text-slate-300'}`}
          >
            <item.icon size={20} />
            <span className="text-[10px] font-medium">{item.label}</span>
          </button>
        ))}
      </div>
    );
  };

  // --- Main Render (Mobile Device Simulation) ---

  return (
    <div className="min-h-screen bg-gray-900 flex items-center justify-center p-4 font-sans selection:bg-indigo-500/30">
        
        {/* Device Frame */}
        <div className="relative w-full max-w-[375px] h-[812px] bg-slate-950 rounded-[40px] shadow-2xl overflow-hidden border-[8px] border-slate-800 ring-1 ring-white/10 flex flex-col">
            
            {/* Notch */}
            <div className="absolute top-0 left-1/2 -translate-x-1/2 w-32 h-6 bg-slate-800 rounded-b-2xl z-[100] flex justify-center items-center">
                <div className="w-12 h-3 bg-slate-900 rounded-full mb-1"></div>
            </div>
            
            {/* Status Bar (Simulated) */}
            <div className="absolute top-1 right-5 z-[100] flex gap-1.5 items-center">
                <Wifi size={12} className="text-white"/>
                <BatteryMedium size={12} className="text-white"/>
            </div>
            <div className="absolute top-1 left-6 z-[100] text-[10px] text-white font-bold">
                9:41
            </div>

            {/* Global Toast - Positioned inside frame */}
            {notification && (
                <div className="absolute top-12 left-1/2 -translate-x-1/2 bg-indigo-600 text-white px-4 py-2 rounded-full shadow-2xl z-[90] animate-bounce whitespace-nowrap text-xs">
                    {notification}
                </div>
            )}

            {/* Main Content Area - Scrollable */}
            <div className="flex-1 overflow-y-auto scrollbar-hide relative bg-slate-950 text-slate-200">
                {loading ? (
                    <div className="h-full flex items-center justify-center">
                        <Loading message={loadingMsg} />
                    </div>
                ) : (
                    <>
                        {view === 'login' && <LoginView onLogin={() => simulateLoading('home', 800)} />}
                        
                        {view === 'home' && (
                        <HomeView 
                            searchQuery={searchQuery}
                            setSearchQuery={setSearchQuery}
                            onSearch={() => simulateLoading('confirmation', 600)}
                            onAiReset={() => {
                            setCustomFeatureQuery("");
                            setSelectedFeatures([]);
                            simulateLoading('features', 600);
                            }}
                        />
                        )}
                        
                        {view === 'confirmation' && (
                        <div className="flex flex-col items-center justify-center h-full p-6 text-center">
                            <Card className="p-8 border-indigo-500/30">
                            <h2 className="text-xl font-bold text-white mb-4">Search Confirmation</h2>
                            <p className="text-slate-300 text-base mb-6">
                                You are looking to compare prices for <br />
                                <span className="text-2xl font-bold text-indigo-400 block mt-2">"{searchQuery}"</span>
                            </p>
                            <div className="flex gap-3 justify-center">
                                <Button variant="secondary" onClick={() => setView('home')} className="text-sm px-4">No, change</Button>
                                <Button onClick={() => simulateLoading('results', 1500)} className="text-sm px-4">Yes</Button>
                            </div>
                            </Card>
                        </div>
                        )}
                        
                        {view === 'results' && (
                        <div className="p-4 pb-24">
                            <div className="flex justify-between items-center mb-6 pt-4">
                                <button onClick={() => setView('home')} className="text-slate-400 hover:text-white flex items-center gap-1"><ChevronLeft size={16}/> Back</button>
                                <h2 className="text-lg font-bold text-white">Price Comparison</h2>
                                <button className="flex items-center gap-1 px-3 py-1.5 bg-slate-800 rounded-lg text-slate-300 hover:bg-slate-700 text-xs">
                                    <Filter size={14} /> Filters
                                </button>
                            </div>

                            <div className="grid gap-4">
                            {MOCK_MOBILES.map((mobile) => (
                                <Card key={mobile.id} className="flex gap-4 hover:bg-slate-800/80 transition-colors group p-3">
                                <div className="w-24 h-24 bg-slate-900 rounded-lg flex-shrink-0 relative overflow-hidden">
                                    <img src={mobile.image} alt={mobile.name} className="w-full h-full object-cover" />
                                    <button 
                                        onClick={(e) => { e.stopPropagation(); toggleFavorite(mobile.id); }}
                                        className={`absolute top-1 right-1 p-1.5 rounded-full ${favorites.includes(mobile.id) ? 'bg-pink-500/20 text-pink-500' : 'bg-slate-800/50 text-slate-400 hover:text-white'}`}
                                    >
                                        <Heart size={14} fill={favorites.includes(mobile.id) ? "currentColor" : "none"} />
                                    </button>
                                </div>
                                
                                <div className="flex-1 flex flex-col justify-between">
                                    <div>
                                        <h3 className="font-bold text-white text-sm leading-tight">{mobile.name}</h3>
                                        <p className="text-xs text-slate-500 mt-0.5">Avg. Market Price</p>
                                        <p className="text-lg font-bold text-indigo-400 mt-0.5">â‚¹{mobile.price.toLocaleString()}</p>
                                        
                                        <div className="flex items-center gap-2 mt-2">
                                            <span className="flex items-center text-yellow-500 text-[10px] gap-0.5"><Star size={10} fill="currentColor"/> {mobile.rating}</span>
                                            <span className={`text-[10px] px-1.5 py-0.5 rounded-full flex items-center gap-0.5 ${mobile.stock ? 'bg-green-500/10 text-green-400' : 'bg-red-500/10 text-red-400'}`}>
                                                {mobile.stock ? <CheckCircle size={8}/> : <AlertCircle size={8}/>}
                                                {mobile.stock ? 'In Stock' : 'Out'}
                                            </span>
                                        </div>
                                    </div>

                                    <div className="flex gap-2 mt-3">
                                        <button 
                                            className="flex-1 bg-indigo-600 hover:bg-indigo-500 text-white py-1.5 rounded text-xs font-medium" 
                                            onClick={() => {
                                                setSelectedProduct(mobile);
                                                setQaResponse("");
                                                setQaInput("");
                                                setReviewSummary("");
                                                simulateLoading('details', 500);
                                            }}
                                        >
                                            View Details
                                        </button>
                                        <button className="flex-1 border border-slate-600 text-slate-300 hover:bg-slate-800 py-1.5 rounded text-xs font-medium">Compare</button>
                                    </div>
                                </div>
                                </Card>
                            ))}
                            </div>
                        </div>
                        )}
                        
                        {view === 'features' && (
                        <FeatureSelectionView 
                            selectedFeatures={selectedFeatures}
                            setSelectedFeatures={setSelectedFeatures}
                            customFeatureQuery={customFeatureQuery}
                            setCustomFeatureQuery={setCustomFeatureQuery}
                            onBack={() => setView('home')}
                            onSubmit={handleAIRecommendation}
                        />
                        )}
                        
                        {view === 'ai-suggestions' && (
                        <div className="p-4 pb-24">
                            <div className="mb-6 pt-4">
                                <button onClick={() => setView('features')} className="text-slate-400 hover:text-white mb-2 flex items-center gap-1"><ChevronLeft size={16}/> Back</button>
                                <h2 className="text-xl font-bold text-white flex items-center gap-2">
                                <Sparkles className="text-teal-400" size={20}/> AI Picks
                                </h2>
                                <p className="text-slate-400 text-xs mt-1">
                                Gemini selected these based on your needs.
                                </p>
                            </div>

                            <div className="grid gap-4">
                            {aiRecommendations.map((mobile) => (
                                <Card key={mobile.id} className="flex flex-col gap-3 group">
                                <div className="h-32 bg-slate-900 rounded-lg overflow-hidden relative flex-shrink-0">
                                    <img src={mobile.image} alt={mobile.name} className="w-full h-full object-cover" />
                                    <div className="absolute top-2 right-2 bg-teal-500 text-white text-[10px] font-bold px-2 py-1 rounded shadow-lg">Gemini Pick</div>
                                </div>
                                
                                <div>
                                    <div className="flex justify-between items-start mb-2">
                                        <div>
                                            <h3 className="font-bold text-white text-base">{mobile.name}</h3>
                                            <p className="text-slate-400 text-xs">{mobile.brand}</p>
                                        </div>
                                        <span className="text-teal-400 font-bold text-sm">â‚¹{mobile.price.toLocaleString()}</span>
                                    </div>
                                    
                                    <div className="bg-teal-500/10 border border-teal-500/20 rounded-lg p-2 mb-3">
                                        <p className="text-teal-300 text-[10px] italic leading-relaxed">
                                        "{mobile.matchReason}"
                                        </p>
                                    </div>

                                    <Button 
                                        variant="outline" 
                                        className="w-full border-teal-500/50 text-teal-400 hover:bg-teal-500/10 text-xs py-2"
                                        onClick={() => {
                                            setSelectedProduct(mobile);
                                            setQaResponse("");
                                            setQaInput("");
                                            simulateLoading('details', 500);
                                        }}
                                    >
                                        See Details
                                    </Button>
                                </div>
                                </Card>
                            ))}
                            </div>
                        </div>
                        )}
                        
                        {view === 'details' && (
                        <DetailsView 
                            product={selectedProduct}
                            onBack={() => setView(aiRecommendations.length > 0 ? 'ai-suggestions' : 'results')}
                            onFavorite={toggleFavorite}
                            isFavorite={favorites.includes(selectedProduct?.id)}
                            qaInput={qaInput}
                            setQaInput={setQaInput}
                            handleProductQA={handleProductQA}
                            qaResponse={qaResponse}
                            qaLoading={qaLoading}
                            reviewSummary={reviewSummary}
                            isGeneratingSummary={isGeneratingSummary}
                            handleReviewSummary={handleReviewSummary}
                            onRequestDemo={startWhatsAppChat}
                            onBuyNow={(retailer) => {
                                setCheckoutRetailer(retailer);
                                simulateLoading('checkout', 1000, `Redirecting to ${retailer}...`);
                            }}
                        />
                        )}

                        {view === 'whatsapp-chat' && (
                        <WhatsAppChatView 
                            store={chatStore}
                            initialMessage={chatMessage}
                            onBack={() => setView('details')}
                            onVideoEnd={() => setView('details')}
                        />
                        )}

                        {view === 'checkout' && (
                            <CheckoutView 
                                retailer={checkoutRetailer}
                                product={selectedProduct}
                                onPlaceOrder={() => simulateLoading('order-success', 1500, "Processing Payment...")}
                                onCancel={() => setView('details')}
                            />
                        )}

                        {view === 'order-success' && (
                            <SuccessView onReturn={() => setView('home')} />
                        )}
                    </>
                )}
            </div>

            {/* Sticky Bottom Nav (Visible on most screens) */}
            <BottomNav />
        </div>
    </div>
  );
}
