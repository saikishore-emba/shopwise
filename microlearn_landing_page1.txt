        function updateCard() {
            const selector = document.getElementById('topicSelector');
            const placeholder = document.getElementById('placeholderState');
            const card = document.getElementById('learningCard');
            const selectedValue = selector.value;

            if (!selectedValue) return;

            // Get Data
            const data = topics[selectedValue];

            // Update DOM Elements
            document.getElementById('cardCategory').innerText = data.category;
            document.getElementById('cardTitle').innerText = data.title;
            document.getElementById('cardSummary').innerText = data.summary;
            document.getElementById('cardTakeaway').innerText = data.takeaway;

            // Build details list
            const detailsContainer = document.getElementById('cardDetails');
            detailsContainer.innerHTML = ''; // Clear previous
            
            data.details.forEach(item => {
                const row = document.createElement('div');
                row.className = "flex flex-col sm:flex-row sm:items-center gap-2 text-sm";
                row.innerHTML = `
                    <span class="${item.color} px-2 py-1 rounded-md font-bold text-xs whitespace-nowrap w-fit">${item.label}</span>
                    <span class="text-gray-700">${item.desc}</span>
                `;
                detailsContainer.appendChild(row);
            });

            // UI Transition
            // 1. Hide placeholder
            placeholder.classList.add('hidden');
            
            // 2. Reset card animation
            card.classList.remove('animate-pop-in');
            card.classList.add('hidden'); // ensure it's hidden to restart
            
            // 3. Trigger reflow
            void card.offsetWidth; 
            
            // 4. Show and animate card
            card.classList.remove('hidden');
            card.classList.add('animate-pop-in');
        }

        // Modal Logic
        const modal = document.getElementById('downloadModal');
        const modalBackdrop = document.getElementById('modalBackdrop');
        const modalPanel = document.getElementById('modalPanel');

        function openModal() {
            modal.classList.remove('hidden');
            // Slight delay to allow display:block to apply before opacity transition
            setTimeout(() => {
                modalBackdrop.classList.remove('opacity-0');
                modalPanel.classList.remove('scale-95', 'opacity-0');
                modalPanel.classList.add('scale-100', 'opacity-100');
            }, 10);
        }

        function closeModal() {
            modalBackdrop.classList.add('opacity-0');
            modalPanel.classList.remove('scale-100', 'opacity-100');
            modalPanel.classList.add('scale-95', 'opacity-0');
            
            setTimeout(() => {
                modal.classList.add('hidden');
            }, 300);
        }
